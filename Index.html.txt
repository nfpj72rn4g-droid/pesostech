<script>
    const totalRows = 10;

    function triggerCamera() {
        document.getElementById('fileInput').click();
    }

    function buildGrid() {
        const cols = parseInt(document.getElementById('colsCount').value);
        const head = document.getElementById('tableHead');
        const body = document.getElementById('tableBody');

        let headHtml = '<th class="row-num">F</th>';
        for(let i=1; i<=cols; i++) { headHtml += `<th>M${i}</th>`; }
        headHtml += '<th>P.Ar</th><th>P.Ab</th>';
        head.innerHTML = headHtml;

        let bodyHtml = '';
        for(let r=1; r<=totalRows; r++) {
            bodyHtml += `<tr><td class="row-num">${r}</td>`;
            for(let c=1; c<=cols; c++) {
                bodyHtml += `
                    <td>
                        <div class="inputs-container">
                            <div class="input-row"><span>Ar</span><input type="number" class="val-u" data-r="${r}" data-c="${c}"></div>
                            <div class="input-row"><span>Ab</span><input type="number" class="val-d" data-r="${r}" data-c="${c}"></div>
                        </div>
                    </td>`;
            }
            bodyHtml += `<td class="row-avg" id="avg-u-${r}">-</td>`;
            bodyHtml += `<td class="row-avg" id="avg-d-${r}">-</td></tr>`;
        }
        body.innerHTML = bodyHtml;
    }

    // Procesamiento OCR inteligente de la imagen
    async function processImage(event) {
        const file = event.target.files[0];
        if (!file) return;

        const status = document.getElementById('statusText');
        status.innerHTML = "⏳ Procesando imagen e identificando números...";
        status.style.color = "#3b82f6";

        try {
            // Inicializar Tesseract para buscar solo números de 3 dígitos (como 200, 300, 430)
            const worker = await Tesseract.createWorker('eng');
            await worker.setParameters({
                tessedit_char_whitelist: '0123456789',
            });
            
            const { data: { text } } = await worker.recognize(file);
            await worker.terminate();

            // Extraer todos los números válidos detectados de 3 dígitos
            const numbers = text.match(/\b\d{3}\b/g);

            if (!numbers || numbers.length === 0) {
                status.innerHTML = "❌ No se detectaron números claros. Intenta con más luz.";
                status.style.color = "#ef4444";
                return;
            }

            // Rellenar automáticamente la cuadrícula alternando Arriba y Abajo
            const inputsUp = document.querySelectorAll('.val-u');
            const inputsDown = document.querySelectorAll('.val-d');
            
            let numIdx = 0;
            const maxCells = Math.min(inputsUp.length, Math.floor(numbers.length / 2));

            // Limpiar datos anteriores
            document.querySelectorAll('input[type="number"]').forEach(i => i.value = '');

            for (let i = 0; i < maxCells; i++) {
                if(numIdx < numbers.length) inputsUp[i].value = numbers[numIdx++];
                if(numIdx < numbers.length) inputsDown[i].value = numbers[numIdx++];
            }

            status.innerHTML = `✅ ¡Escaneo exitoso! Se cargaron ${maxCells * 2} registros.`;
            status.style.color = var(--accent);
            
            // Ejecutar cálculos automáticamente después de cargar los datos
            runCalculations();

        } catch (error) {
            status.innerHTML = "❌ Error al procesar la foto.";
            status.style.color = "#ef4444";
            console.error(error);
        }
    }

    function runCalculations() {
        const cols = parseInt(document.getElementById('colsCount').value);
        let sUp = 0, cUp = 0, sDown = 0, cDown = 0;

        for(let r=1; r<=totalRows; r++) {
            let rSUp = 0, rCUp = 0, rSDown = 0, rCDown = 0;

            for(let c=1; c<=cols; c++) {
                const inUp = document.querySelector(`.val-u[data-r="${r}"][data-c="${c}"]`);
                const inDn = document.querySelector(`.val-d[data-r="${r}"][data-c="${c}"]`);

                const vU = parseFloat(inUp.value);
                const vD = parseFloat(inDn.value);

                if(!isNaN(vU)) { rSUp += vU; rCUp++; sUp += vU; cUp++; }
                if(!isNaN(vD)) { rSDown += vD; rCDown++; sDown += vD; cDown++; }
            }

            document.getElementById(`avg-u-${r}`).innerText = rCUp > 0 ? (rSUp/rCUp).toFixed(1) : '-';
            document.getElementById(`avg-d-${r}`).innerText = rCDown > 0 ? (rSDown/rCDown).toFixed(1) : '-';
        }

        const finalUp = cUp > 0 ? (sUp/cUp) : 0;
        const finalDown = cDown > 0 ? (sDown/cDown) : 0;
        const totalElements = cUp + cDown;
        const finalTotal = totalElements > 0 ? ((sUp + sDown)/totalElements) : 0;

        document.getElementById('resUp').innerText = finalUp > 0 ? finalUp.toFixed(2) : '-';
        document.getElementById('resDown').innerText = finalDown > 0 ? finalDown.toFixed(2) : '-';
        document.getElementById('resTotal').innerText = finalTotal > 0 ? finalTotal.toFixed(2) : '-';
    }

    window.onload = buildGrid;
</script>
